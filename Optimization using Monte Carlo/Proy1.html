set.seed(2018)
#Problema:
#Minimizar x^2+y^2-6xy
#S.a (x,y)>=0
#   x+y<=1
#   3x+y<=1.5

n <- (10 ^ 3)


fmin <- 1000 #valor muy grande
x1 <- 99
y1 <- 99 #valores arbitrarios


x<-NULL
y<-NULL
X<-NULL
Y<-NULL
X1<-NULL
Y1<-NULL
for (i in seq(n)) {
  x[i] <- (runif(1) * .5)
  y[i]<- (runif(1))*(1-x[i])
  if (x[i] + y[i] <= 1) { #primera restricciÃ³n
    if (3 * x[i] + y[i] <= 1.5) { #segunda restricciÃ³n
      X<-c(X,x[i])
      Y<-c(Y,y[i])
      if (x[i] ^ 2 + y[i] ^ 2 - 6 * x[i] * y[i] < fmin) { #funciÃ³n objetivo
        fmin <- x[i] ^ 2 + y[i] ^ 2 - 6 * x[i] * y[i]
        x1 <- x[i]
        y1 <- y[i]
        X1<-c(X1,x1)
        Y1<-c(Y1,y1)
        
      }
    }
  }
}

cat('valor minimo de x:',x1,';')

        X1<-c(X1,x1)

        y1 <- y[2*i]
        X1<-c(X1,x1)
